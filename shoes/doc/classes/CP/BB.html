static VALUE
rb_cpBBInitialize(VALUE self, VALUE l, VALUE b, VALUE r, VALUE t)
{
        cpBB *bb = BBGET(self);
        bb->l = NUM2DBL(l);
        bb->b = NUM2DBL(b);
        bb->r = NUM2DBL(r);
        bb->t = NUM2DBL(t);
        
        return self;
}

static VALUE
rb_cpBBToString(VALUE self)
{
        char str[256];
        cpBB *bb = BBGET(self);
        
        sprintf(str, "#<CP::BB:(% .3f, % .3f) -> (% .3f, % .3f)>", bb->l, bb->b, bb->r, bb->t);
        
        return rb_str_new2(str);
}
