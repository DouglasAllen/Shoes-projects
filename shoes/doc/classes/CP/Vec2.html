static VALUE
rb_cpVectNear(VALUE self, VALUE v, VALUE d)
{
        cpFloat dist = NUM2DBL(d);
        cpVect delta = cpvsub(*VGET(self), *VGET(v));
        return (cpvdot(delta, delta) <= dist*dist) ? Qtrue : Qfalse;
}
