def / len
  a = []
  each_with_index do |x, i|
    a << [] if i % len == 0
    a.last << x
  end
  a
end

def inspect(hits = {})
    return "[...]" if hits[self]
    hits[self] = true
    "[" + map { |x| x.method(:inspect).arity == 0 ? x.inspect : x.inspect(hits) }.join(', ') + "]"
end

def to_html
  ary = self
  Web.Bit {
    h5 "A List of Things"
    h1 "An Array"
    unless ary.empty?
      ol {
        ary.map { |x| li { self << HTML(x) } }
      }
    end
  }
end
