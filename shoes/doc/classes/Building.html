def initialize(west, east, north, south)
        @west, @east, @north, @south = west, east, north, south
        @top, @bottom = 1.1 + rand(3) * 0.15, 1.0
        
        color = (1..3).collect { 0.2 + 0.4 * rand }
        color << 0.9
        @stroke = $app.rgb *color
        color[-1] = 0.3
        @fill = $app.rgb *color
end
