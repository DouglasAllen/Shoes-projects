def run(str)
  obj = nil
  @io << str
  @io.rewind
  unless l = lex
    raise Empty if @line == ''
  else
    case l.strip
    when "reset"
      @line = ""
    when "time"
      @line = "puts %{You started \#{IRBalike.started.since} ago.}"
    else
      @line << l << "\n"
      if @ltype or @continue or @indent > 0
        raise Continue
      end
    end
  end
  unless @line.empty?
    obj = eval @line, TOPLEVEL_BINDING, "(irb)", @line_no
  end
  @line_no += @line.scan(/\n/).length
  @line = ''
  @exp_line_no = @line_no

  @indent = 0
  @indent_stack = []

  $stdout.rewind
  output = $stdout.read
  $stdout.truncate(0)
  $stdout.rewind
  [output, obj]
rescue Object => e
  case e when Empty, Continue
  else @line = ""
  end
  raise e
ensure
  set_input(StringIO.new)
end
