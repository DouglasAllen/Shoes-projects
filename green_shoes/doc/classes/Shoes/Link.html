def initialize str, arg, &blk
  @link_proc = if blk
    blk
  elsif arg[:click].is_a? String
    proc{Shoes.APPS.first.app.visit arg[:click]}
  elsif arg[:click].nil?
    proc{}
  else
    arg[:click]
  end

  @pos, @index, @link_hover = nil, nil, false
  super str
end
